490  static size_t varint_build(int64_t val, uint8_t* buf) noexcept {
491  buf[0] = val & 0x7f;
492  val >>= 7;
493  int i = 1;

497  val >>= 7;

711  template <typename T> result message_field(int64_t field_id, const T& msg) noexcept {
712  if (m_error != result::ok) return m_error;
713  // This gives us a worst case estimate of the blob size
714  auto size = msg.estimate_size();
715  if (size == 0) size = SIZE_MAX;
716  auto dummy_size = encoder::varint_size(size);
717  uint8_t dummy_varint[10] = {};
718  // We write the object header
719  m_error = m_encoder.field_header(field_id, wire_type::length_blob);
720  if (m_error != result::ok) return m_error;
721  // note down the current position
722  auto pos = m_encoder.stream().position();
723  // write a dummy varint based on the estimated size
724  m_error = m_encoder.fixed(dummy_varint, dummy_size);
725  if (m_error != result::ok) return m_error;
726  // and hand of encoding to the message type
727  m_error = msg.encode(*this);
728  if (m_error != result::ok) return m_error;
729  // after it is done, we calculate the size difference
730  auto real_size = m_encoder.stream().position() - (pos + dummy_size);
731  if (real_size > size) return minipb::result::general_error;
732  // Build our real size and patch it to the dummy size
733  encoder::varint_build(real_size, dummy_varint);
734  for (size_t i = 0; i < dummy_size - 1; i++)
735  dummy_varint[i] |= 0x80;
736  // Patch out our dummy size
737  m_error = m_encoder.stream().write_at(pos, dummy_varint, dummy_size);
738  return m_error;
739  }

892  result varint(uint64_t& val) noexcept {
893  uint8_t buf[10]{};
894  val = 0;
895  auto peek_size = m_stream.peek(buf, sizeof(buf));
896  if (peek_size == 0) { // Peek unsupported or no data
897  for (size_t i = 0; i < 10; i++) {
898  auto res = m_stream.read(&buf[0], 1);
899  if (res != result::ok) return res;
900  val |= static_cast<uint64_t>(buf[i] & 0x7f) << (i * 7);
901  if ((buf[i] & 0x80) == 0) return res;
902  }
903  } else {
904  for (size_t i = 0; i < peek_size; i++) {
905  val |= static_cast<uint64_t>(buf[i] & 0x7f) << (i * 7);
906  if ((buf[i] & 0x80) == 0) {
907  return m_stream.skip(i + 1);
908  }
909  }
910  }
911 
912  return result::invalid_input;
